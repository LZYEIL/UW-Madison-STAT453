class Perceptron():
    def __init__(self, num_features):
        self.weights = [0.0 for _ in range(num_features)]
        self.bias = 0.0

    def forward(self, x):
        linear = sum(w_i * x_i for w_i, x_i in zip(self.weights, x)) + self.bias
        prediction = 1 if linear >= 0 else 0
        return prediction

    def backward(self, x, y):
        prediction = self.forward(x)

        error = y - prediction
        self.bias += error
        self.weights = [w_i + error * x_i for w_i, x_i in zip(self.weights, x)] 
        

    def train(self, x, y, epochs):
        for e in range(epochs):

            for i in range(len(y)):
                self.backward(x[i], y[i])

    

    def evaluate(self, x, y):
        correct = 0
        for i in range(len(y)):
            pred = self.forward(x[i])
            if pred == y[i]:
                correct += 1
        accuracy = correct / len(y)

        return accuracy
